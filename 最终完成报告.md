## 🚀 使用示例

### 完整游戏流程

```python
from cultivation import Player, World, GameEngine
from cultivation.utils.config import Config
from cultivation.system.battle_system import BattleSystem
from cultivation.system.skill_system import SkillSystem
from cultivation.system.quest_system import QuestSystem
from cultivation.system.sect_system import SectSystem
from cultivation.system.alchemy_system import AlchemySystem
from cultivation.system.economy_system import EconomySystem
from cultivation.core.event_system import EventSystem

# 初始化
config = Config()
event_system = EventSystem()
player = Player(name="张三")
world = World()

# 战斗系统
battle_system = BattleSystem(event_system)
enemy = {"name": "妖兽", "hp": 100, "attack": 15}
result = battle_system.start_battle(player, enemy)

# 技能系统
skill_system = SkillSystem()
skill_system.learn_skill(player, "fireball")
skill_system.use_skill(player, "fireball", enemy)

# 任务系统
quest_system = QuestSystem(event_system)
quest_system.accept_quest(player, "quest_001")

# 门派系统
sect_system = SectSystem(event_system)
sect_system.join_sect(player, "qingyun")

# 炼丹系统
alchemy_system = AlchemySystem(event_system)
alchemy_system.learn_recipe(player, "huichun")
pill = alchemy_system.refine_pill(player, "huichun", {"灵草": 2})

# 经济系统
economy_system = EconomySystem()
economy_system.buy_item(player, "lingshi", 10)

# 成就系统（自动追踪）
# 完成一定数量行动后自动解锁
```
